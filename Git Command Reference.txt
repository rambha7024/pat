Git Command Reference

# 🔥 1. git init (Create Repository)

```bash
git init                       # Init repo here
git init my-project            # Create + init repo
git init --initial-branch=main # Init with main branch

🔥 2. git config

# Set identity (one-time setup)
git config --global user.name "Name"      # Set username
git config --global user.email "Email"    # Set email

git config --list                         # View config
git config --global --edit                # Edit config manually

commands to delete Git config 👇

git config --unset user.name            # Delete username (current repo)
git config --unset user.email           # Delete email (current repo)

git config --global --unset user.name   # Delete global username
git config --global --unset user.email  # Delete global email

🔥 3. git status
git status                     # Show repo status
git status --short             # Short status output
git status --branch            # Show branch info
git status --ignored           # Show ignored files
git status -uno                # Hide untracked files

🔥 4. git add
git add file.txt               # Add specific file
git add .                      # Add all files
git add --force file.txt       # Force add ignored file

🔥 5. git commit
git commit -m "message"        # Commit with message
git commit --verbose           # Show changes while committing
git commit --allow-empty -m "" # Empty commit


🔥 6. git log
git log                        # Full commit history
git log --oneline              # One line per commit
git log --graph                # Show branch graph
git log --decorate             # Show branch/tag names
git log --all                  # All branches history
git log --pretty=oneline       # Custom one-line format


🔥 7. git show
git show                       # Show last commit details
git show HEAD                  # Show current commit
git show <commit-id>           # Show specific commit
git show HEAD~1                # Show previous commit
git show --pretty              # Custom output
git show --pretty=oneline      # One-line format
git show <id>:file.txt         # Show file from commit


🔥 8. git diff
git diff                       # Unstaged changes
git diff HEAD                  # Working vs last commit
git diff HEAD~1                # Compare previous commit
git diff file.txt              # Changes in file
git diff branch1 branch2       # Compare branches
git diff --staged              # Staged changes


🔥 9. git branch
git branch                     # List branches
git branch branch-name         # Create branch
git branch -a                  # All branches
git branch -d branch-name      # Delete branch
git branch -D branch-name      # Force delete
git branch -m new-name         # Rename branch


🔥 switch (modern)
git switch branch-name          # Switch branch (modern way)
git switch -c new-branch        # Create + switch branch
git switch -C branch-name       # Force create/reset branch
git switch -                    # Go to previous branch


🔥 10. git checkout
git checkout branch-name        # Switch branch
git checkout -b new-branch      # Create + switch branch
git checkout file.txt           # Restore file from last commit
git checkout HEAD~1 file.txt    # Restore file from previous commit
git checkout -- .               # Restore all files
git checkout -- file.txt        # Restore one file

🔥 11. git merge
git merge branch-name           # Merge branch into current
git merge --ff-only branch-name # Only fast-forward, else fail
git merge --squash branch-name  # Combine commits into one
git merge --abort               # Cancel merge
git merge --continue            # Continue after conflict


🔥 12. git rebase
git rebase branch-name          # Move commits onto another branch
git rebase --continue           # Continue after conflict
git rebase --abort              # Cancel rebase
git rebase --autosquash         # Auto squash commits


🔥 13. git reset
git reset HEAD~1                # Undo last commit (keep changes)
git reset --soft HEAD~1         # Undo commit, keep staged
git reset --hard HEAD~1         # Delete commit + changes
git reset file.txt              # Unstage file


🔥 14. git restore
git restore file.txt            # Restore file
git restore .                   # Restore all files
git restore --staged file.txt   # Unstage file

🔥 15. git revert
git revert <commit-id>          # Undo commit safely
git revert HEAD                 # Undo last commit


🔥 16. git remote
git remote                      # List remotes
git remote -v                   # Show remote URLs
git remote add origin URL       # Add remote
git remote remove origin        # Remove remote
git remote rename old new       # Rename remote
git remote show origin          # Remote details


🔥 17. git clone
git clone URL                   # Copy repo from remote

🔥 18. git fetch
git fetch                       # Download changes (no merge)
git fetch origin                # Fetch from origin
git fetch origin branch         # Fetch specific branch


🔥 19. git pull
git pull                        # Fetch + merge
git pull origin main            # Pull specific branch
git pull --rebase               # Pull with rebase
git pull --no-rebase            # Pull with merge
git pull --ff-only              # Only fast-forward



🔥 20. git push
git push                        # Push current branch
git push origin main            # Push to remote branch
git push -u origin main         # Set upstream + push
git push --set-upstream origin main # Same as -u
git push --force                # Force push
git push --tags                 # Push tags
git push origin --delete branch-name # Delete remote branch


🔥 21. git stash
git stash                       # Save changes temporarily
git stash push                  # Same as stash
git stash push -m "msg"         # Stash with message
git stash -u                    # Include untracked files
git stash list                  # Show stashes
git stash show                  # Show stash changes
git stash show -p               # Detailed changes
git stash apply                 # Apply stash (keep it)
git stash pop                   # Apply + remove stash
git stash drop                  # Delete stash
git stash clear                 # Delete all stashes
git stash branch new-branch     # Create branch from stash


🔥 22. git tag
git tag                         # List tags
git tag v1.0                    # Create tag
git tag -a v1.0 -m "version"    # Annotated tag
git tag -d v1.0                 # Delete tag
git push origin v1.0            # Push tag
git push --tags                 # Push all tags
git show v1.0                   # Tag details


🔥 23. git rm / mv
git rm file.txt                 # Delete file + stage
git rm -f file.txt              # Force delete
git mv old.txt new.txt          # Rename file


🔥 24. git clean
git clean -f                    # Delete untracked files
git clean -i                    # Interactive clean

🔥 25. git blame
git blame file.txt              # Who changed each line
git blame -L 10,20 file.txt     # Specific lines
git blame -e file.txt           # Show email

🔥 26. git reflog
git reflog                      # History of HEAD changes
git reflog expire               # Clean old reflog
git reset --hard HEAD@{1}       # Recover previous state

🔥 27. git cherry-pick
git cherry-pick <commit>        # Apply commit to current branch
git cherry-pick --abort         # Cancel cherry-pick
git cherry-pick --continue      # Continue after conflict
      
🔥 MOST IMPORTANT MERGE FLAGS
git merge --no-ff feature       # Keep merge commit
git merge --ff-only feature     # Only fast-forward
git merge --squash feature      # One combined commit
🔥 Interview Gold (Super Important)
Command	Meaning	Safe
merge	Combine branches	✅
rebase	Rewrite history	⚠️
reset --hard	Delete history	❌
revert	Safe undo commit	✅

EXTRAS:
🔥 git commit --amend (Modify Last Commit)

Used to edit the most recent commit (message or content).

git commit --amend

➡ Opens editor to change commit message.

git commit --amend -m "new message"

➡ Replace last commit message directly.

git commit --amend --no-edit

➡ Add staged changes to last commit without changing message.

✅ Common Uses:

Fix typo in commit message

Add forgotten file to last commit

Clean commit before push

⚠️ Avoid amending commits that are already pushed (unless force push).
